36       void _insert_iterative(const myType &value) {
37           Edge *newNode = new Edge(value);
38           if (rootTree == nullptr) {
39               rootTree = newNode;
40               is_inserted = true;
41               return;
42           }
43           Edge *current = rootTree;
44           Edge *parent = nullptr;
45           while (current) {
46               parent = current;
47               if (value == current->value) {
48                   // duplicate not allowed
49                   is_inserted = false;
50                   return;
51               }
52               if (value < current->value) {
53                   current = current->left;
54               } else if (value > current->value) {
55                   current = current->right;
56               }
57           }
58           if (value < parent->value) {
59               parent->left = newNode;
60           } else {
61               parent->right = newNode;
62           }
63           is_inserted = true;
64       }

75           if (value < root->value) {
76               root->left = _insert_recursive(root->left, value);
77               return root;
78           } else {
79               root->right = _insert_recursive(root->right, value);
80               return root;
81           }
82       }

85           if (rootTree == nullptr) {
86               rootTree = new Edge(value);
87               return;
88           }
89           queue<Edge *> q;
90           q.push(rootTree);
91           while (!q.empty()) {
92               Edge *current = q.front();
93               q.pop();
94               if (!current->left) {
95                   current->left = new Edge(value);
96                   return;
97               } else {
98                   q.push(current->left);
99               }
100              if (!current->right) {
101                  current->right = new Edge(value);
102                  return;
103              } else {
104                  q.push(current->right);
105              }
106          }

144          Edge *left_node = node->left;
145          Edge *right_of_left_node = left_node->right;

169      Edge *_insert_using_avl_function_recursive(Edge *root, const myType &value) {
170          if (root == nullptr) {
171              return new Edge(value);
172          }
173          if (value < root->value) {
174              root->left = _insert_using_avl_function_recursive(root->left, value);
175          } else if (value > root->value) {
176              root->right = _insert_using_avl_function_recursive(root->right, value);
177          } else {
178              // duplicates are not allowed
179              is_inserted = false;
180              return root;
181          }
182          root->height = 1 + _max(_height(root->left), _height(root->right));
183          int balance = _get_balance(root);
184  
185          // there are 4 cases
186  
187          // left left case
188          if (balance > 1 and value < root->left->value) {
189              return _right_rotate(root);
190          }
191  
192          // right right case
193          if (balance < -1 and value > root->right->value) {
194              return _left_rotate(root);
195          }
196  
197          // left right case
198          if (balance > 1 and value > root->left->value) {
199              root->left = _left_rotate(root->left);
200              return _right_rotate(root);
201          }
202  
203          // right left case
204          if (balance < -1 and value < root->right->value) {
205              root->right = _right_rotate(root->right);
206              return _left_rotate(root);
207          }
208  
209          return root;
210      }

212      Edge *_delete_value_and_get_successor_from_right_subtree_recursive(Edge *root, const myType &value) {
213          if (root == nullptr) {
214              // value not found
215              is_deleted = false;
216              return root;
217          }
218          if (value < root->value) {
219              root->left = _delete_value_and_get_successor_from_right_subtree_recursive(root->left, value);
220          } else if (value > root->value) {
221              root->right = _delete_value_and_get_successor_from_right_subtree_recursive(root->right, value);
222          } else {
223              // node with only one child or no child
224              if (root->left == nullptr or root->right == nullptr) {
225                  Edge *successor = root->left ? root->left : root->right;
226                  if (successor == nullptr) {
227                      // root is leaf
228                      delete root;
229                      delete successor;
230                      root = nullptr;
231                  } else {
232                      // root has one child
233                      *root = *successor; // copy content temp to root
234                  }
235              } else {
236                  // node with tow children
237                  Edge *successor = _min_value_in_right_sub_tree(root->right);
238                  root->value = successor->value;
239                  root->right = _delete_value_and_get_successor_from_right_subtree_recursive(root->right, root->value);
240              }
241          }
242          // If the tree had only one node
243          if (root == nullptr) {
244              return root;
245          }
246  
247          root->height = 1 + _max(_height(root->left), _height(root->right));
248          int balance = _get_balance(root);
249  
250          // there are 4 cases
251  
252          // left left case
253          if (balance > 1 and _get_balance(root->left) >= 0) {
254              return _right_rotate(root);
255          }
256  
257          // left right case
258          if (balance > 1 and _get_balance(root->left) < 0) {
259              root->left = _left_rotate(root->left);
260              return _right_rotate(root);
261          }
262  
263          // right right case
264          if (balance < -1 and _get_balance(root->right) <= 0) {
265              return _left_rotate(root);
266          }
267  
268          // right left case
269          if (balance < -1 and _get_balance(root->right) > 0) {
270              root->right = _right_rotate(root->right);
271              return _left_rotate(root);
272          }
273  
274          return root;
275      }

355          Edge *parent = nullptr;
356          Edge *current = rootTree;
357          // search for node to delete it
358          while (current) {
359              if (current->value == value) {
360                  break;
361              }
362              parent = current;
363              if (value < current->value) {
364                  current = current->left;
365              } else {
366                  current = current->right;
367              }
368          }
369          if (current == nullptr) {
370              // value not found
371              is_deleted = false;
372              return;
373          }
374          // node with only one child or no child
375          if (current->left == nullptr or current->right == nullptr) {
376              Edge *successor = current->left ? current->left : current->right;
377              if (successor == nullptr) {
378                  // node is leaf, no children
379                  delete successor;
380                  if (current == rootTree) {
381                      delete rootTree;
382                      rootTree = nullptr;
383                  } else if (current == parent->left) {
384                      delete parent->left;
385                      parent->left = nullptr;
386                  } else {
387                      delete parent->right;
388                      parent->right = nullptr;
389                  }
390              } else {
391                  // node has one child
392                  *current = *successor; // copy content successor to current
393                  delete successor;
394              }
395          } else {
396              // If the node to be deleted has two children
397              _delete_and_get_node_from_right_sub_tree(current);
398  //            _delete_and_get_node_from_left_sub_tree(current);
399          }
400          is_deleted = true;
401      }

404          // find successor from right sub tree
405          Edge *parent = nullptr;
406          Edge *successor = current->right;
407          while (successor->left) {
408              parent = successor;
409              successor = successor->left;
410          }
411          current->value = successor->value;
412          if (parent) {
413              // maybe successor->right: nullptr or it has child from right
414              parent->left = successor->right;
415          } else {
416              current->right = successor->right;
417          }
418          delete successor;
419      }

573      bool _is_complete_iterative() {
574          queue<Edge *> q;
575          q.emplace(rootTree);
576          bool check_to_pointer_is_null = false;
577          while (!q.empty()) {
578              Edge *current = q.front();
579              q.pop();
580              if (check_to_pointer_is_null and current) {
581                  return false;
582              }
583              if (current) {
584                  q.emplace(current->left);
585                  q.emplace(current->right);
586              }
587              if (!current) {
588                  check_to_pointer_is_null = true;
589              }
590          }
591          return true;
592      }

722          int max_degree = 0;
723          int current_degree;
724          while (!q.empty()) {

726              Edge *node = q.front();
727              q.pop();
728              if (node->left) {
729                  q.emplace(node->left);
730                  current_degree++;
731              }

733                  q.emplace(node->right);
734                  current_degree++;
735              }
736              max_degree = height(max_degree, current_degree);

738                  break;
739              }

811      // Auxiliary Space: O(1)
812      static int _power(int x, unsigned n) {
813          int result = 1;
814          while (n) {
815              if (n & 1) {
816                  result = result * x;
817              }
818              x = x * x;
819              n >>= 1;
820          }
821          return result;
822      }

862      void _print_hierarchically_iterative() {
863          int w = 3;
864          int count_level = _count_levels_from_node_to_bottom_recursive(rootTree);
865          int i = 0;
866          queue<Edge *> q;
867          q.emplace(rootTree);
868          while (!q.empty()) {
869              if (i == count_level) {
870                  break;
871              }
872              cout << "l" << setw(2) << i << '.';
873              int len = q.size();
874              int first = _power(2, count_level - i - 1) - 1;
875              first *= SPACE;
876              _print_space(first);
877              int between = _power(2, count_level - i);
878              between *= SPACE;
879              between -= w;
880              i++;
881              while (len--) {
882                  Edge *current = q.front();
883                  q.pop();
884                  if (current) {
885                      cout << setw(w) << current->value;
886                  } else {
887                      // N : Null
888                      cout << setw(w) << "N";
889                  }
890                  if (current and current->left) {
891                      q.emplace(current->left);
892                  } else if (i < count_level) {
893                      q.emplace(nullptr);
894                  }
895                  if (current and current->right) {
896                      q.emplace(current->right);
897                  } else if (i < count_level) {
898                      q.emplace(nullptr);
899                  }
900                  if (len > 0) {
901                      // don't print after last value in the level
902                      _print_space(between);
903                  }
904              }
905              cout << endl;
906          }
907      }
