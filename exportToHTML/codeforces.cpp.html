28           int c = s[0] - 'a';
29           pre[c][0]++;
30           for (int i = 1; i < n; ++i) {
31               c = s[i] - 'a';
32               for (int j = 0; j < 26; ++j) {
33                   pre[j][i] = pre[j][i - 1];
34                   pre[j][i] += (c == j);

39               int l, r;

44               l--;
45               if (r < n) {
46                   if (l > 0) {
47                       cout << pre[c][r] - pre[c][l - 1] << endl;
48                   } else {
49                       cout << pre[c][r] << endl;

51               } else {
52                   int ans = 0;
53                   int al = l % n;
54                   int ar = r % n;
55                   if (al > 0) {
56                       ans += (pre[c][n - 1] - pre[c][al - 1]);
57                   }
58                   ans += pre[c][ar];
59                   l += al;
60                   r -= ar;
61                   int between = r - l;
62                   if (between >= n) {
63                       between /= n;
64                       ans += (between * pre[c][n - 1]);
65                   }
66                   cout << ans << endl;
67               }
68           }
