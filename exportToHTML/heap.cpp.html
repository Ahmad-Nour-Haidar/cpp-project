11   /// Heap Sort: Time Complexity = (n * log(n))
12   /// Heap Sort: Space = Only space of array
13   template<class T>
14   class HeapSort {
15       /* This class only for sorting array 
16        * Note: values store in array from index 0 so: 
17        * left child  = 2*i+1 
18        * right child = 2*i+2 
19        * */
20   private:
21   
22       void build_max_heap(T *array, int length) {
23           for (int i = length / 2 - 1; i >= 0; --i) {
24               max_heapify(array, length, i);
25           }
26       }
27   
28       void max_heapify(T *array, int length, int index) {
29           int left = 2 * index + 1;
30           int right = 2 * index + 2;
31           int largest = index;
32           if (left < length and array[left] > array[largest]) {
33               largest = left;
34           }
35           if (right < length and array[right] > array[largest]) {
36               largest = right;
37           }
38           if (largest not_eq index) {
39               _swap(array[index], array[largest]);
40               max_heapify(array, length, largest);
41           }
42       }
43   
44       void build_min_heap(T *array, int length) {
45           for (int i = length / 2 - 1; i >= 0; --i) {
46               min_heapify(array, length, i);
47           }
48       }
49   
50       void min_heapify(T *array, int length, int index) {
51           int left = 2 * index + 1;
52           int right = 2 * index + 2;
53           int largest = index;
54           if (left < length and array[left] < array[largest]) {
55               largest = left;
56           }
57           if (right < length and array[right] < array[largest]) {
58               largest = right;
59           }
60           if (largest not_eq index) {
61               _swap(array[index], array[largest]);
62               min_heapify(array, length, largest);
63           }
64       }
65   
66       void _swap(T &a, T &b) {
67           T temp = a;
68           a = b;
69           b = temp;
70       }
71   
72   public:
73       void descending_heap_sort(T *array, int length) { // height to min
74           build_min_heap(array, length);
75           for (int i = length - 1; i >= 0; --i) {
76               _swap(array[0], array[i]);
77               min_heapify(array, i, 0);
78           }
79       }
80   
81       void ascending_heap_sort(T *array, int length) { // min to height
82           build_max_heap(array, length);
83           for (int i = length - 1; i >= 0; --i) {
84               _swap(array[0], array[i]);
85               max_heapify(array, i, 0);
86           }
87       }
88   
89       void print(T *array, int length) {
90           for (int i = 0; i < length; ++i) {
91               cout << array[i] << ' ';
92           }
93           cout << endl;
94       }
95   };

97   int main() {
98   
99       HeapSort<int> h;
100      int a[] = {1, 5, 48, 57, 986, 32, 0, 14, -9, 6, 5, 4, 7, -9, 32, 14, 52, 698, 47, 14, 21, -9, -88, -96, -85, -21};
101      int len = sizeof(a) / sizeof(a[0]);
102  
103      h.print(a, len);
104  
105      h.ascending_heap_sort(a, len);
106      h.print(a, len);
107  
108      h.descending_heap_sort(a, len);
109      h.print(a, len);
110  
111      return 0;
