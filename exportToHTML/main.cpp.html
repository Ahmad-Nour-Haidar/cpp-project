7    void insert_value(int &counter, int &n, int *&a, int value) {
8        if (counter < n) {
9            a[counter] = value;
10           counter++;
11       } else {
12           n = 2 * n; // new size to array
13           int *b = new int[n];
14           for (int i = 0; i < counter; ++i) {
15               b[i] = a[i]; // copy elements
16           }
17           b[counter] = value;
18           counter++;
19           a = b;
20       }
21   }

25       int n, counter;
